# Phase 3.3: Integration Testing Plan

## Overview

This document specifies testing strategies, test cases, and validation procedures for the agentic tokenomics system.

> **Scope Note:** The smart contract unit tests in this document primarily cover the
> **M008 Attestation Bond** mechanism (BondStatus lifecycle, challenge flows, slashing).
> **M010 Reputation Signal** tests are covered separately in
> `mechanisms/m010-reputation-signal/reference-impl/` with test vectors and fixtures.
> Full M010 signal lifecycle tests (active → withdrawn → challenged → invalidated)
> are planned for a future iteration.

---

## Testing Strategy

### Test Pyramid

```
                    ┌─────────┐
                    │   E2E   │  ← 10% of tests
                    │  Tests  │     Full system validation
                   ─┴─────────┴─
                  ┌─────────────┐
                  │ Integration │  ← 30% of tests
                  │   Tests     │     Component interactions
                 ─┴─────────────┴─
                ┌─────────────────┐
                │    Unit Tests   │  ← 60% of tests
                │                 │     Individual functions
               ─┴─────────────────┴─
```

### Test Categories

| Category | Scope | Tools | Coverage Target |
|----------|-------|-------|-----------------|
| Unit | Functions, modules | Jest, Go test | 80% |
| Integration | Component interactions | Jest, Testcontainers | 70% |
| Contract | Smart contracts | CosmWasm test framework | 90% |
| E2E | Full workflows | Playwright, Custom harness | Critical paths |
| Load | Performance | k6, Locust | SLO validation |
| Security | Vulnerability scanning | Slither, Semgrep | All contracts |

        attestation_iri: "regen:13abc...".to_string(),
        beneficiary: None,
    };

    app.execute_contract(
        Addr::unchecked("attester"),
        attestation_addr.clone(),
        &create_msg,
        &[Coin::new(100_000_000u128, "uregen")],

    .unwrap();

import { check, sleep } from "k6";

const errorRate = new Rate("errors");
const responseTime = new Trend("response_time");

export const options = {
  stages: [
    { duration: "1m", target: 10 },   // Ramp up
    { duration: "5m", target: 50 },   // Sustain
    { duration: "1m", target: 100 },  // Peak
    { duration: "1m", target: 0 },    // Ramp down
  ],
  thresholds: {
    http_req_duration: ["p(95)<2000"], // 95% under 2s
    errors: ["rate<0.1"],              // Error rate under 10%
  },
};

const BASE_URL = __ENV.AGENT_URL || "http://localhost:8080";

export default function () {
  // Simulate governance analysis request
  const governancePayload = JSON.stringify({
    message: "Analyze proposal 62",
    agentId: "governance-analyst",
  });

  const governanceRes = http.post(`${BASE_URL}/api/agent/message`, governancePayload, {
    headers: { "Content-Type": "application/json" },
  });

  check(governanceRes, {
    "governance status 200": (r) => r.status === 200,
    "governance has response": (r) => r.json().response !== undefined,
  });

  errorRate.add(governanceRes.status !== 200);
  responseTime.add(governanceRes.timings.duration);

  sleep(1);

  // Simulate market health check
  const marketPayload = JSON.stringify({
    message: "Market health check",
    agentId: "market-monitor",

  });

// tests/integration/economic_reboot_test.go

package integration_test

import (
    "testing"
    "time"

    "github.com/stretchr/testify/suite"

type EconomicRebootTestSuite struct {
    suite.Suite
    app *SimApp
}

func (s *EconomicRebootTestSuite) TestFullEconomicRebootLifecycle() {
    // Phase 1: Activate M013 Fee Routing
    s.Run("activate_fee_routing", func() {
        s.submitGovernanceProposal(MsgUpdateFeeConfig{
            Authority: govModuleAddr,
            FeeConfig: defaultFeeConfig(),
        })
        state := s.app.FeeRouterKeeper.GetModuleState(s.ctx)
        s.Equal("TRANSITION", state.Phase)
    })

    // Phase 2: Execute credit transactions and verify fees
    s.Run("collect_fees", func() {
        s.executeCreditIssuance(100) // 100 REGEN value
        records := s.app.FeeRouterKeeper.GetFeeRecords(s.ctx, 1)
        s.Len(records, 1)
        s.Equal("issuance", records[0].TxType)
        s.True(records[0].FeeCollected.Amount.IsPositive())
    })

    // Phase 3: Activate M014 PoA
    s.Run("activate_poa", func() {
        // Apply and approve 15 validators (5 per category)
        for i := 0; i < 5; i++ {
            s.applyAndApproveValidator("infrastructure_builder", i)
            s.applyAndApproveValidator("refi_partner", i)
            s.applyAndApproveValidator("data_steward", i)
        }

        state := s.app.AuthorityKeeper.GetModuleState(s.ctx)
        s.Equal(uint32(15), state.ActiveValidatorCount)
    })

    // Phase 4: Activate M012 Dynamic Supply
    s.Run("activate_supply", func() {
        s.submitGovernanceProposal(MsgActivateSupplyModule{
            Authority: govModuleAddr,
            HardCap: "221000000000000",
            BaseRegrowthRate: 200,
            PeriodLengthSeconds: 604800,
        })
        state := s.app.SupplyKeeper.GetSupplyState(s.ctx)
        s.Equal("TRANSITION", state.Phase)
    })

    // Phase 5: Advance time and verify mint/burn cycle
    s.Run("mint_burn_cycle", func() {
        s.advanceTime(7 * 24 * time.Hour) // One period
        s.app.EndBlocker(s.ctx, abci.RequestEndBlock{})

        state := s.app.SupplyKeeper.GetSupplyState(s.ctx)
        s.True(state.LastMint != "0" || state.LastBurn != "0")
    })

    // Phase 6: Activate M015 Rewards
    s.Run("activate_rewards", func() {
        s.submitGovernanceProposal(MsgActivateRewards{
            Authority: govModuleAddr,
            Params: defaultRewardParams(),
        })
        state := s.app.RewardsKeeper.GetModuleState(s.ctx)
        s.Equal("TRACKING", state.Phase)
    })

    // Phase 7: Advance through tracking period
    s.Run("tracking_to_distributing", func() {
        s.advanceTime(90 * 24 * time.Hour) // 3 months

        // Generate activity
        s.executeCreditPurchase("buyer1", 50)
        s.executeCreditRetirement("retiree1", 30)
        s.castGovernanceVote("voter1", 1, "yes")

        s.app.EndBlocker(s.ctx, abci.RequestEndBlock{})

        state := s.app.RewardsKeeper.GetModuleState(s.ctx)
        s.Equal("DISTRIBUTING", state.Phase)
    })
}

func TestEconomicRebootSuite(t *testing.T) {
    suite.Run(t, new(EconomicRebootTestSuite))
}
