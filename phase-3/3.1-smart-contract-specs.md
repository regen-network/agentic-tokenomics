# Phase 3.1: Smart Contract Development Specifications

## Overview

This document provides implementation-ready specifications for smart contracts supporting the agentic tokenomics system. Contracts are divided between native Cosmos SDK module extensions and CosmWasm smart contracts.

```protobuf
  uint64 proposal_id = 2;

  uint32 score = 3;
```

### Keeper Implementation

```go
// x/ecocredit/keeper/msg_server_class_creator.go

package keeper

import (
    "context"
    sdk "github.com/cosmos/cosmos-sdk/types"
    "github.com/regen-network/regen-ledger/x/ecocredit/v3/base"
)

// ProposeClassCreator handles MsgProposeClassCreator
func (k Keeper) ProposeClassCreator(
    ctx context.Context,
    msg *base.MsgProposeClassCreator,
) (*base.MsgProposeClassCreatorResponse, error) {
    sdkCtx := sdk.UnwrapSDKContext(ctx)

    // Validate credit type exists
    if !k.creditTypeExists(ctx, msg.CreditType) {
        return nil, ErrInvalidCreditType
    }

    // Validate deposit meets minimum
    minDeposit := k.GetClassCreatorMinDeposit(ctx)
    if !msg.Deposit.IsAllGTE(minDeposit) {
        return nil, ErrInsufficientDeposit
    }

    // Lock deposit in escrow
    proposer, err := sdk.AccAddressFromBech32(msg.Proposer)
    if err != nil {
        return nil, err
    }
    if err := k.bankKeeper.SendCoinsFromAccountToModule(
        ctx, proposer, base.ModuleName, msg.Deposit,
    ); err != nil {
        return nil, err
    }

    // Create proposal
    proposalID := k.getNextProposalID(ctx)
    proposal := &base.ClassCreatorProposal{
        Id:             proposalID,
        Proposer:       msg.Proposer,
        CreditType:     msg.CreditType,
        MethodologyIri: msg.MethodologyIri,
        AdminAddress:   msg.AdminAddress,
        Status:         "PENDING_REVIEW",
        Deposit:        msg.Deposit,
        SubmitTime:     sdkCtx.BlockTime(),
    }

    k.setClassCreatorProposal(ctx, proposal)

    // Emit event
    sdkCtx.EventManager().EmitEvent(
        sdk.NewEvent(
            "class_creator_proposed",
            sdk.NewAttribute("proposal_id", fmt.Sprintf("%d", proposalID)),
            sdk.NewAttribute("proposer", msg.Proposer),
            sdk.NewAttribute("credit_type", msg.CreditType),
        ),
    )

    return &base.MsgProposeClassCreatorResponse{
        ProposalId: proposalID,
    }, nil
}

// SubmitAgentScore handles MsgSubmitAgentScore
func (k Keeper) SubmitAgentScore(
    ctx context.Context,
    msg *base.MsgSubmitAgentScore,
) (*base.MsgSubmitAgentScoreResponse, error) {
    sdkCtx := sdk.UnwrapSDKContext(ctx)

    // Verify agent is authorized
    if !k.isAuthorizedAgent(ctx, msg.Agent) {
        return nil, ErrUnauthorizedAgent
    }

    // Get proposal
    proposal, err := k.getClassCreatorProposal(ctx, msg.ProposalId)
    if err != nil {
        return nil, err
    }

    // Verify proposal is in correct state
    if proposal.Status != "PENDING_REVIEW" {
        return nil, ErrInvalidProposalStatus
    }

    // Record agent score
    proposal.AgentScore = &base.AgentScore{
        Agent:         msg.Agent,
        Score:         msg.Score,
        Confidence:    msg.Confidence,
        Recommendation: msg.Recommendation,
        RationaleIri:  msg.RationaleIri,
        ScoredAt:      sdkCtx.BlockTime(),
    }
    proposal.Status = "AGENT_REVIEWED"

    k.setClassCreatorProposal(ctx, proposal)

    // Auto-promote to governance if score exceeds configurable thresholds
    // These values should be governance parameters, not hardcoded
    const AUTO_PROMOTE_SCORE_THRESHOLD = 700       // minimum agent score (0-1000)
    const AUTO_PROMOTE_CONFIDENCE_THRESHOLD = 850  // minimum confidence (0-1000)
    if msg.Score >= AUTO_PROMOTE_SCORE_THRESHOLD && msg.Confidence >= AUTO_PROMOTE_CONFIDENCE_THRESHOLD {
        k.promoteToGovernance(ctx, proposal)
    }

    return &base.MsgSubmitAgentScoreResponse{}, nil
}
```

### Core Contract Logic

```rust
// contracts/attestation-bond/src/contract.rs

use cosmwasm_std::{
    entry_point, to_binary, BankMsg, Binary, Coin, CosmosMsg, Deps, DepsMut,
    Env, MessageInfo, Response, StdResult, Uint128,
};

use crate::error::ContractError;
use crate::msg::{ExecuteMsg, InstantiateMsg, QueryMsg};
use crate::state::{
    Attestation, AttestationType, BondStatus, Challenge, ChallengeStatus,
    Config, ATTESTATIONS, ATTESTATION_TYPES, CHALLENGES, CONFIG,
    NEXT_ATTESTATION_ID, NEXT_CHALLENGE_ID,
};

#[entry_point]
pub fn execute(
    deps: DepsMut,
    env: Env,
    info: MessageInfo,
    msg: ExecuteMsg,
) -> Result<Response, ContractError> {
    match msg {
        ExecuteMsg::CreateAttestation {
            attestation_type,
            attestation_iri,
            beneficiary,
        } => execute_create_attestation(
            deps, env, info, attestation_type, attestation_iri, beneficiary,
        ),
        ExecuteMsg::ChallengeAttestation {
            attestation_id,
            evidence_iri,
        } => execute_challenge_attestation(deps, env, info, attestation_id, evidence_iri),
        ExecuteMsg::ResolveChallenge {
            challenge_id,
            resolution,
        } => execute_resolve_challenge(deps, env, info, challenge_id, resolution),
        ExecuteMsg::ReleaseBond { attestation_id } => {
            execute_release_bond(deps, env, info, attestation_id)
        }
        // ... other handlers
    }
}

fn execute_create_attestation(
    deps: DepsMut,
    env: Env,
    info: MessageInfo,
    attestation_type: String,
    attestation_iri: String,
    beneficiary: Option<String>,
) -> Result<Response, ContractError> {
    // Load attestation type config
    let att_type = ATTESTATION_TYPES
        .load(deps.storage, &attestation_type)
        .map_err(|_| ContractError::UnknownAttestationType {})?;

    // Validate bond amount
    let bond = info
        .funds
        .iter()
        .find(|c| c.denom == "uregen")
        .ok_or(ContractError::NoBondProvided {})?;

    if bond.amount < att_type.min_bond {
        return Err(ContractError::InsufficientBond {
            required: att_type.min_bond,
            provided: bond.amount,
        });
    }

    // Calculate timestamps
    let lock_expires = env.block.time.plus_days(att_type.lock_period_days);
    let challenge_closes = env.block.time.plus_days(att_type.challenge_window_days);

    // Create attestation
    let id = NEXT_ATTESTATION_ID.load(deps.storage).unwrap_or(1);
    let attestation = Attestation {
        id,
        attester: info.sender.clone(),
        attestation_type,
        attestation_iri,
        beneficiary: beneficiary.map(|b| deps.api.addr_validate(&b)).transpose()?,
        bond: bond.clone(),
        status: BondStatus::Active,
        bonded_at: env.block.time,
        activated_at: None,
        lock_expires_at: lock_expires,
        challenge_window_closes_at: challenge_closes,
    };

    ATTESTATIONS.save(deps.storage, id, &attestation)?;
    NEXT_ATTESTATION_ID.save(deps.storage, &(id + 1))?;

    Ok(Response::new()
        .add_attribute("action", "create_attestation")
        .add_attribute("attestation_id", id.to_string())
        .add_attribute("attester", info.sender)
        .add_attribute("bond_amount", bond.amount))
}

fn execute_challenge_attestation(
    deps: DepsMut,
    env: Env,
    info: MessageInfo,
    attestation_id: u64,
    evidence_iri: String,
) -> Result<Response, ContractError> {
    let config = CONFIG.load(deps.storage)?;
    let mut attestation = ATTESTATIONS.load(deps.storage, attestation_id)?;

    // Verify attestation can be challenged
    if attestation.status != BondStatus::Active {
        return Err(ContractError::AttestationNotChallengeable {});
    }

    if env.block.time > attestation.challenge_window_closes_at {
        return Err(ContractError::ChallengeWindowClosed {});
    }

    // Validate challenge deposit (10% of bond)
    let min_deposit = attestation.bond.amount
        .multiply_ratio(config.min_challenge_deposit_ratio, Uint128::from(BASIS_POINTS_DIVISOR));

    let deposit = info
        .funds
        .iter()
        .find(|c| c.denom == attestation.bond.denom)
        .ok_or(ContractError::NoDepositProvided {})?;

    if deposit.amount < min_deposit {
        return Err(ContractError::InsufficientChallengeDeposit {});
    }

    // Create challenge
    let challenge_id = NEXT_CHALLENGE_ID.load(deps.storage).unwrap_or(1);
    let challenge = Challenge {
        id: challenge_id,
        attestation_id,
        challenger: info.sender.clone(),
        evidence_iri,
        deposit: deposit.clone(),
        status: ChallengeStatus::Pending,
        created_at: env.block.time,
        resolved_at: None,
        resolution: None,
    };

    CHALLENGES.save(deps.storage, challenge_id, &challenge)?;
    NEXT_CHALLENGE_ID.save(deps.storage, &(challenge_id + 1))?;

    // Update attestation status
    attestation.status = BondStatus::Challenged;
    ATTESTATIONS.save(deps.storage, attestation_id, &attestation)?;

    // Notify arbiter DAO (via submessage to DAO DAO)
    let notify_msg = create_arbiter_proposal_msg(&config.arbiter_dao, &challenge)?;

    Ok(Response::new()
        .add_message(notify_msg)
        .add_attribute("action", "challenge_attestation")
        .add_attribute("challenge_id", challenge_id.to_string())
        .add_attribute("attestation_id", attestation_id.to_string())
        .add_attribute("challenger", info.sender))
}

fn execute_resolve_challenge(
    deps: DepsMut,
    env: Env,
    info: MessageInfo,
    challenge_id: u64,
    resolution: ChallengeResolutionInput,
) -> Result<Response, ContractError> {
    let config = CONFIG.load(deps.storage)?;

    // Only arbiter DAO can resolve
    if info.sender != config.arbiter_dao {
        return Err(ContractError::Unauthorized {});
    }

    let mut challenge = CHALLENGES.load(deps.storage, challenge_id)?;
    let mut attestation = ATTESTATIONS.load(deps.storage, challenge.attestation_id)?;

    // Calculate distributions
    let arbiter_fee = attestation.bond.amount
        .multiply_ratio(config.arbiter_fee_ratio, Uint128::from(BASIS_POINTS_DIVISOR));

    let mut messages: Vec<CosmosMsg> = vec![];

    match resolution {
        ChallengeResolutionInput::AttesterWins => {
            // Attester gets bond + challenge deposit - arbiter fee
            let attester_amount = attestation.bond.amount
                .checked_add(challenge.deposit.amount)?
                .checked_sub(arbiter_fee)?;

            messages.push(CosmosMsg::Bank(BankMsg::Send {
                to_address: attestation.attester.to_string(),
                amount: vec![Coin {
                    denom: attestation.bond.denom.clone(),
                    amount: attester_amount,
                }],
            }));

            attestation.status = BondStatus::ResolvedValid;
            challenge.resolution = Some(ChallengeResolution::AttesterWins);
        }
        ChallengeResolutionInput::ChallengerWins => {
            // Challenger gets 50% of bond + deposit - arbiter fee
            let challenger_amount = attestation.bond.amount
                .multiply_ratio(Uint128::from(50u128), Uint128::from(100u128))
                .checked_add(challenge.deposit.amount)?
                .checked_sub(arbiter_fee)?;

            // Community pool gets remaining 50%
            let community_amount = attestation.bond.amount
                .multiply_ratio(Uint128::from(50u128), Uint128::from(100u128));

            messages.push(CosmosMsg::Bank(BankMsg::Send {
                to_address: challenge.challenger.to_string(),
                amount: vec![Coin {
                    denom: attestation.bond.denom.clone(),
                    amount: challenger_amount,
                }],
            }));

            // Send to community pool (via distribution module)
            // This requires a custom message to x/distribution

            attestation.status = BondStatus::Slashed;
            challenge.resolution = Some(ChallengeResolution::ChallengerWins);
        }
    }

    // Send arbiter fee
    messages.push(CosmosMsg::Bank(BankMsg::Send {
        to_address: config.arbiter_dao.to_string(),
        amount: vec![Coin {
            denom: attestation.bond.denom.clone(),
            amount: arbiter_fee,
        }],
    }));

    challenge.status = ChallengeStatus::Resolved;
    challenge.resolved_at = Some(env.block.time);

    CHALLENGES.save(deps.storage, challenge_id, &challenge)?;
    ATTESTATIONS.save(deps.storage, challenge.attestation_id, &attestation)?;

    Ok(Response::new()
        .add_messages(messages)
        .add_attribute("action", "resolve_challenge")
        .add_attribute("challenge_id", challenge_id.to_string())
        .add_attribute("resolution", format!("{:?}", resolution)))
}
```

```rust
fn execute_submit_signal(
    deps: DepsMut,
    env: Env,
    info: MessageInfo,
    subject_type: SubjectType,
    subject_id: String,
    category: String,
    endorsement_level: u8,
    rationale_iri: Option<String>,
) -> Result<Response, ContractError> {
    let config = CONFIG.load(deps.storage)?;

    // Validate endorsement level
    if endorsement_level < 1 || endorsement_level > 5 {
        return Err(ContractError::InvalidEndorsementLevel {});
    }

    // Query signaler's staked amount from x/staking
    let stake = query_staked_amount(deps.as_ref(), &info.sender)?;

    // Verify minimum stake for category
    let min_stake = get_min_stake_for_category(&config, &category)?;
    if stake < min_stake {
        return Err(ContractError::InsufficientStake {
            required: min_stake,
            actual: stake,
        });
    }

    // Create signal
    let signal_id = NEXT_SIGNAL_ID.load(deps.storage).unwrap_or(1);
    let signal = ReputationSignal {
        id: signal_id,
        signaler: info.sender.clone(),
        subject_type: subject_type.clone(),
        subject_id: subject_id.clone(),
        category: category.clone(),
        endorsement_level,
        rationale_iri,
        stake_at_signal: stake,
        created_at: env.block.time,
        status: SignalStatus::Active,
    };

    SIGNALS.save(deps.storage, signal_id, &signal)?;
    NEXT_SIGNAL_ID.save(deps.storage, &(signal_id + 1))?;

    // Update indexes
    update_signal_indexes(deps.storage, &signal)?;

    // Recompute aggregated reputation
    recompute_aggregated_reputation(
        deps.storage,
        &env,
        &config,
        &subject_type,
        &subject_id,
        &category,
    )?;

    Ok(Response::new()
        .add_attribute("action", "submit_signal")
        .add_attribute("signal_id", signal_id.to_string())
        .add_attribute("subject", format!("{:?}:{}", subject_type, subject_id))
        .add_attribute("category", category)
        .add_attribute("endorsement", endorsement_level.to_string()))
}

fn recompute_aggregated_reputation(
    storage: &mut dyn Storage,
    env: &Env,
    config: &Config,
    subject_type: &SubjectType,
    subject_id: &str,
    category: &str,
) -> StdResult<()> {
    let subject_key = format!("{:?}", subject_type);
    let signal_ids = SIGNALS_BY_SUBJECT
        .may_load(storage, (&subject_key, subject_id))?
        .unwrap_or_default();

    let mut total_weighted_score = Uint128::zero();
    let mut total_weight = Uint128::zero();
    let mut signal_count = 0u32;

    for signal_id in signal_ids {
        let signal = SIGNALS.load(storage, signal_id)?;

        if signal.status != SignalStatus::Active || signal.category != category {
            continue;
        }

        // Calculate time decay
        let age_seconds = env.block.time.seconds() - signal.created_at.seconds();
        let half_life_seconds = config.decay_half_life_days * 86400;
        let decay_factor = calculate_decay(age_seconds, half_life_seconds);

        // Weight = stake * decay
        let weight = signal.stake_at_signal.multiply_ratio(decay_factor, 1000u128);

        // Score contribution = weight * endorsement_level
        let score_contribution = weight.multiply_ratio(
            Uint128::from(signal.endorsement_level as u128),
            Uint128::from(5u128),  // Normalize to 0-1
        );

        total_weighted_score = total_weighted_score.checked_add(score_contribution)?;
        total_weight = total_weight.checked_add(weight)?;
        signal_count += 1;
    }

    // Compute final score (0-1000)
    let score = if total_weight.is_zero() {
        Uint128::zero()
    } else {
        total_weighted_score.multiply_ratio(1000u128, total_weight)
    };

    let aggregated = AggregatedReputation {
        subject_type: subject_type.clone(),
        subject_id: subject_id.to_string(),
        category: category.to_string(),
        score,
        signal_count,
        last_updated: env.block.time,
    };

    AGGREGATED.save(storage, (&subject_key, subject_id, category), &aggregated)?;

    Ok(())
}

fn calculate_decay(age_seconds: u64, half_life_seconds: u64) -> u128 {
    // Approximate exponential decay: e^(-age/half_life)
    // Using: 1000 * 2^(-age/half_life) ≈ 1000 * (1/2)^(age/half_life)
    let half_lives = age_seconds / half_life_seconds;
    if half_lives >= 10 {
        return 1;  // Minimum decay factor
    }
    1000u128 >> half_lives  // Bit shift approximation
}
```

### Overview

```go
type Keeper struct {
    cdc        codec.Codec
    storeKey   storetypes.StoreKey
    bankKeeper banktypes.Keeper
    authority  string // governance module address
}

// CollectFee is the core fee interception logic, called as a hook
// from x/ecocredit message handlers.
func (k Keeper) CollectFee(
    ctx sdk.Context,
    feePayer sdk.AccAddress,
    txType string,
    transactionValue sdk.Coin,
) (sdk.Coin, error) {
    config := k.GetFeeConfig(ctx)
    shares := k.GetDistributionShares(ctx)
    state := k.GetModuleState(ctx)

    // In FLAT_GAS phase, no value-based fees are collected
    if state.Phase == "FLAT_GAS" {
        return sdk.NewCoin(transactionValue.Denom, sdk.ZeroInt()), nil
    }

    // Determine fee rate based on transaction type
    var feeRateBps uint32
    switch txType {
    case "issuance":
        feeRateBps = config.IssuanceFeeRate
    case "transfer":
        feeRateBps = config.TransferFeeRate
    case "retirement":
        feeRateBps = config.RetirementFeeRate
    case "trade":
        feeRateBps = config.TradeFeeRate
    default:
        return sdk.Coin{}, fmt.Errorf("unknown transaction type: %s", txType)
    }

    // Safety: enforce max fee rate
    if feeRateBps > config.MaxFeeRateBps {
        feeRateBps = config.MaxFeeRateBps
    }

    // Calculate fee amount
    feeAmount := transactionValue.Amount.
        Mul(sdk.NewIntFromUint64(uint64(feeRateBps))).
        Quo(sdk.NewInt(10000))

    // Enforce minimum fee floor
    minFee := config.MinFee.Amount
    if feeAmount.LT(minFee) {
        feeAmount = minFee
    }

    feeCoin := sdk.NewCoin(transactionValue.Denom, feeAmount)

    // Transfer fee from payer to module account
    if err := k.bankKeeper.SendCoinsFromAccountToModule(
        ctx, feePayer, types.ModuleName, sdk.NewCoins(feeCoin),
    ); err != nil {
        return sdk.Coin{}, fmt.Errorf("fee collection failed: %w", err)
    }

    // Distribute to pools
    if err := k.distributeFee(ctx, feeCoin, shares); err != nil {
        return sdk.Coin{}, fmt.Errorf("fee distribution failed: %w", err)
    }

    // Record fee event
    k.recordFee(ctx, feePayer, txType, transactionValue, feeCoin, shares)

    ctx.EventManager().EmitEvent(
        sdk.NewEvent(
            "fee_collected",
            sdk.NewAttribute("payer", feePayer.String()),
            sdk.NewAttribute("tx_type", txType),
            sdk.NewAttribute("value", transactionValue.String()),
            sdk.NewAttribute("fee", feeCoin.String()),
        ),
    )

    return feeCoin, nil
}

// distributeFee splits the collected fee across the four pools.
func (k Keeper) distributeFee(
    ctx sdk.Context,
    fee sdk.Coin,
    shares types.DistributionShares,
) error {
    total := fee.Amount

    burnAmt := total.Mul(sdk.NewIntFromUint64(uint64(shares.BurnShare))).
        Quo(sdk.NewInt(10000))
    validatorAmt := total.Mul(sdk.NewIntFromUint64(uint64(shares.ValidatorShare))).
        Quo(sdk.NewInt(10000))
    communityAmt := total.Mul(sdk.NewIntFromUint64(uint64(shares.CommunityShare))).
        Quo(sdk.NewInt(10000))
    // Agent infra gets the remainder to avoid rounding loss
    agentAmt := total.Sub(burnAmt).Sub(validatorAmt).Sub(communityAmt)

    // Burn pool: send to x/supply module account for deferred burning.
    // x/supply's ExecuteMintBurn calls BurnCoins at period boundaries,
    // ensuring burn is accounted for in the mint/burn algorithm (M012).
    // x/feerouter does NOT burn directly — this avoids a double-accounting
    // contradiction where M012 would miss burn events.
    if burnAmt.IsPositive() {
        burnCoins := sdk.NewCoins(sdk.NewCoin(fee.Denom, burnAmt))
        if err := k.bankKeeper.SendCoinsFromModuleToModule(
            ctx, types.ModuleName, supplytypes.ModuleName, burnCoins,
        ); err != nil {
            return err
        }
        // x/supply accumulates this via GetAndResetPeriodBurnAccumulator
        k.supplyKeeper.AccumulateBurn(ctx, burnAmt)
    }

    // Validator fund: send to validator fund module account
    if validatorAmt.IsPositive() {
        valCoins := sdk.NewCoins(sdk.NewCoin(fee.Denom, validatorAmt))
        if err := k.bankKeeper.SendCoinsFromModuleToModule(
            ctx, types.ModuleName, types.ValidatorFundName, valCoins,
        ); err != nil {
            return err
        }
    }

    // Community pool: use x/distribution FundCommunityPool
    if communityAmt.IsPositive() {
        communityCoins := sdk.NewCoins(sdk.NewCoin(fee.Denom, communityAmt))
        moduleAddr := authtypes.NewModuleAddress(types.ModuleName)
        if err := k.bankKeeper.SendCoins(
            ctx, moduleAddr, authtypes.NewModuleAddress(distrtypes.ModuleName),
            communityCoins,
        ); err != nil {
            return err
        }
    }

    // Agent infrastructure fund: send to agent fund module account
    if agentAmt.IsPositive() {
        agentCoins := sdk.NewCoins(sdk.NewCoin(fee.Denom, agentAmt))
        if err := k.bankKeeper.SendCoinsFromModuleToModule(
            ctx, types.ModuleName, types.AgentInfraFundName, agentCoins,
        ); err != nil {
            return err
        }
    }

    return nil
}
```

```go
    if refPrice.IsZero() {
        return sdk.NewCoin("uregen", sdk.ZeroInt())
    }
    value := refPrice.Mul(amount.TruncateInt())
    return sdk.NewCoin("uregen", value)
}
```

```go
// x/supply/keeper/mint_burn.go

package keeper

import (
    "math"

    sdk "github.com/cosmos/cosmos-sdk/types"
    stakingtypes "github.com/cosmos/cosmos-sdk/x/staking/types"

    "github.com/regen-network/regen-ledger/x/supply/types"
)

// ExecuteMintBurn performs the per-period supply adjustment.
// Called from EndBlocker when a period boundary is crossed.
// Returns (minted, burned, newSupply, error) so the caller and
// MsgExecuteMintBurnResponse can report all three values.
func (k Keeper) ExecuteMintBurn(ctx sdk.Context) (sdk.Int, sdk.Int, sdk.Int, error) {
    params := k.GetSupplyParams(ctx)
    state := k.GetSupplyState(ctx)

    if state.Phase == "INFLATIONARY" {
        return sdk.ZeroInt(), sdk.ZeroInt(), sdk.ZeroInt(), nil
    }

    hardCap, ok := sdk.NewIntFromString(params.HardCap)
    if !ok {
        return sdk.ZeroInt(), sdk.ZeroInt(), sdk.ZeroInt(), types.ErrInvalidHardCap
    }

    currentSupply, ok := sdk.NewIntFromString(state.CurrentSupply)
    if !ok {
        return sdk.ZeroInt(), sdk.ZeroInt(), sdk.ZeroInt(), types.ErrInvalidSupply
    }

    // --- Compute Minting (Regrowth) ---
    // M[t] = r × (C - S[t])
    // where r = r_base × effective_multiplier × ecological_multiplier

    headroom := hardCap.Sub(currentSupply)
    if headroom.IsNegative() {
        headroom = sdk.ZeroInt() // Safety: cap inviolability
    }

    // Base regrowth rate (basis points → decimal)
    rBase := sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.BaseRegrowthRate))).
        Quo(sdk.NewDec(10000))

    // Effective multiplier: depends on M014 phase
    effectiveMultiplier := k.computeEffectiveMultiplier(ctx, params, state)

    // Ecological multiplier (v0: disabled = 1.0)
    ecologicalMultiplier := sdk.OneDec()
    if params.EcologicalMultiplierEnabled {
        ecologicalMultiplier = k.computeEcologicalMultiplier(ctx, params)
        if ecologicalMultiplier.IsNegative() {
            ecologicalMultiplier = sdk.ZeroDec() // Floor at 0 per spec
        }
    }

    // Combined regrowth rate
    r := rBase.Mul(effectiveMultiplier).Mul(ecologicalMultiplier)

    // Minted = r × headroom
    mintedDec := r.MulInt(headroom)
    minted := mintedDec.TruncateInt()

    // --- Compute Burning ---
    // B[t] is accumulated from M013 burn pool during the period.
    // x/feerouter sends burn-designated tokens to x/supply's module account;
    // we now execute the actual burn here, keeping burn accounting atomic
    // with the mint/burn cycle.
    burned := k.GetAndResetPeriodBurnAccumulator(ctx)
    if burned.IsPositive() {
        burnCoins := sdk.NewCoins(sdk.NewCoin("uregen", burned))
        if err := k.bankKeeper.BurnCoins(ctx, types.ModuleName, burnCoins); err != nil {
            return sdk.ZeroInt(), sdk.ZeroInt(), currentSupply, err
        }
    }

    // --- Apply ---
    if minted.IsPositive() {
        mintCoins := sdk.NewCoins(sdk.NewCoin("uregen", minted))
        if err := k.bankKeeper.MintCoins(ctx, types.ModuleName, mintCoins); err != nil {
            return sdk.ZeroInt(), sdk.ZeroInt(), currentSupply, err
        }
        // Minted tokens go to the community pool for M015 distribution
        moduleAddr := k.accountKeeper.GetModuleAddress(types.ModuleName)
        if err := k.distrKeeper.FundCommunityPool(
            ctx, mintCoins, moduleAddr,
        ); err != nil {
            return sdk.ZeroInt(), sdk.ZeroInt(), currentSupply, err
        }
    }

    // Update state
    newSupply := currentSupply.Add(minted).Sub(burned)

    // Safety: cap inviolability
    if newSupply.GT(hardCap) {
        excess := newSupply.Sub(hardCap)
        minted = minted.Sub(excess)
        newSupply = hardCap
    }

    // Safety: non-negative supply
    if newSupply.IsNegative() {
        newSupply = sdk.ZeroInt()
    }

    // Check equilibrium
    diff := minted.Sub(burned).Abs()
    equilibriumThreshold := hardCap.Quo(sdk.NewInt(10000)) // 0.01% of cap
    if diff.LT(equilibriumThreshold) {
        state.ConsecutiveEquilibriumPeriods++
    } else {
        state.ConsecutiveEquilibriumPeriods = 0
    }

    // Phase transitions
    if state.Phase == "TRANSITION" && burned.IsPositive() {
        state.Phase = "DYNAMIC"
    }
    if state.Phase == "DYNAMIC" && state.ConsecutiveEquilibriumPeriods >= 12 {
        state.Phase = "EQUILIBRIUM"
    }

    state.CurrentSupply = newSupply.String()
    state.LastMint = minted.String()
    state.LastBurn = burned.String()
    state.LastPeriodEnd = ctx.BlockTime()
    k.SetSupplyState(ctx, state)

    // Record history
    k.recordMintBurn(ctx, currentSupply, newSupply, minted, burned,
        rBase, effectiveMultiplier, ecologicalMultiplier)

    // Emit events
    ctx.EventManager().EmitEvents(sdk.Events{
        sdk.NewEvent("supply_mint",
            sdk.NewAttribute("amount", minted.String()),
            sdk.NewAttribute("regrowth_rate", r.String()),
        ),
        sdk.NewEvent("supply_burn",
            sdk.NewAttribute("amount", burned.String()),
        ),
        sdk.NewEvent("supply_state",
            sdk.NewAttribute("supply", newSupply.String()),
            sdk.NewAttribute("phase", state.Phase),
        ),
    })

    return minted, burned, newSupply, nil
}

// computeEffectiveMultiplier selects the right multiplier based on M014 phase.
//
// Phase-gated behavior per spec:
//   - Pre-M014 (INACTIVE):   staking_multiplier only
//   - M014 Phase 1-2 (TRANSITION): max(staking, stability)
//   - M014 Phase 3+ (ACTIVE/EQUILIBRIUM): stability_multiplier only
func (k Keeper) computeEffectiveMultiplier(
    ctx sdk.Context,
    params types.SupplyParams,
    state types.SupplyState,
) sdk.Dec {
    stakingMul := sdk.OneDec()
    stabilityMul := sdk.OneDec()

    if params.StakingMultiplierEnabled {
        bondedRatio := k.stakingKeeper.BondedRatio(ctx)
        // staking_multiplier = 1 + (S_staked / S_total), range [1.0, 2.0]
        stakingMul = sdk.OneDec().Add(bondedRatio)
        if stakingMul.GT(sdk.NewDec(2)) {
            stakingMul = sdk.NewDec(2)
        }
    }

    if params.StabilityMultiplierEnabled {
        stabilityRatio := k.rewardsKeeper.StabilityCommitmentRatio(ctx)
        // stability_multiplier = 1 + (S_stability / S_total), range [1.0, 2.0]
        stabilityMul = sdk.OneDec().Add(stabilityRatio)
        if stabilityMul.GT(sdk.NewDec(2)) {
            stabilityMul = sdk.NewDec(2)
        }
    }

    switch state.M014Phase {
    case "INACTIVE":
        return stakingMul
    case "TRANSITION":
        // max(staking, stability) prevents regrowth discontinuity
        if stakingMul.GT(stabilityMul) {
            return stakingMul
        }
        return stabilityMul
    default: // "ACTIVE", "EQUILIBRIUM"
        return stabilityMul
    }
}
```

### Security Invariants

1. **Cap Inviolability**: `current_supply <= hard_cap` at all times; EndBlocker enforces this after every mint
2. **Non-Negative Supply**: `current_supply >= 0`; burn cannot reduce below zero
3. **Monotonic Cap**: `hard_cap` can only be changed via Layer 4 governance (67% supermajority)
4. **Mint-Burn Independence**: Minting and burning are computed independently; neither can block the other
5. **Parameter Bound Safety**: `base_regrowth_rate ∈ [0, 1000]` (0-10%); enforced on parameter update
6. **Period Atomicity**: Each period's mint/burn executes atomically in EndBlocker; partial execution is not possible

```go
// ApplyValidator handles MsgApplyValidator
func (k Keeper) ApplyValidator(
    ctx sdk.Context,
    msg *types.MsgApplyValidator,

    // Validate category
    if !types.IsValidCategory(msg.Category) {
        return nil, fmt.Errorf("invalid category: %s (must be infrastructure_builder, refi_partner, or data_steward)", msg.Category)
    }

    }

    if k.isActiveValidator(ctx, msg.OperatorAddress) {
        return nil, types.ErrAlreadyActive
    }

    appID := k.getNextApplicationID(ctx)
    app := types.ValidatorApplication{
        Id:               appID,
        Applicant:        msg.Applicant,
        OperatorAddress:  msg.OperatorAddress,
        Category:         msg.Category,
        OrganizationName: msg.OrganizationName,

        SubmittedAt:      ctx.BlockTime(),

            sdk.NewAttribute("application_id", fmt.Sprintf("%d", appID)),
            sdk.NewAttribute("operator", msg.OperatorAddress),
            sdk.NewAttribute("category", msg.Category),
        ),
    )

    return &types.MsgApplyValidatorResponse{ApplicationId: appID}, nil

// ApproveValidator handles MsgApproveValidator
func (k Keeper) ApproveValidator(
    ctx sdk.Context,
    msg *types.MsgApproveValidator,
) error {
    params := k.GetParams(ctx)
    state := k.GetModuleState(ctx)

    app, err := k.getApplication(ctx, msg.ApplicationId)
    if err != nil {
        return err
    }

    if app.Status != "PENDING" {
        return types.ErrInvalidApplicationStatus
    }

    // Check if slot is available
    if state.ActiveValidatorCount >= params.MaxValidators {
        return types.ErrMaxValidatorsReached
    }

    // Calculate term end
    termDuration := time.Duration(params.TermLengthDays) * 24 * time.Hour
    termEnd := msg.TermStart.Add(termDuration)

    // Create authority validator
    validator := types.AuthorityValidator{
        OperatorAddress:  app.OperatorAddress,
        ApplicantAddress: app.Applicant,
        Category:         app.Category,
        OrganizationName: app.OrganizationName,
        EvidenceIri:      app.EvidenceIri,
        Status:           "APPROVED",
        TermStart:        msg.TermStart,
        TermEnd:          termEnd,
        JoinedAt:         ctx.BlockTime(),
        Performance:      &types.PerformanceRecord{},
    }

    k.setAuthorityValidator(ctx, &validator)

    // Update application status
    app.Status = "APPROVED"
    app.DecidedAt = ctx.BlockTime()
    k.setApplication(ctx, app)

    // Update category counts
    k.incrementCategoryCount(ctx, app.Category)

    ctx.EventManager().EmitEvent(
        sdk.NewEvent("validator_approved",
            sdk.NewAttribute("operator", app.OperatorAddress),
            sdk.NewAttribute("category", app.Category),
            sdk.NewAttribute("term_end", termEnd.String()),
        ),
    )

    return nil
}

// ActivateValidator transitions APPROVED → ACTIVE once the node is operational.
// Called when the validator's signing key appears in the Tendermint validator set.
func (k Keeper) ActivateValidator(ctx sdk.Context, operatorAddr string) error {
    validator, err := k.getAuthorityValidator(ctx, operatorAddr)
    if err != nil {
        return err
    }

    if validator.Status != "APPROVED" {
        return types.ErrInvalidStatus
    }

    validator.Status = "ACTIVE"
    k.setAuthorityValidator(ctx, validator)

    state := k.GetModuleState(ctx)
    state.ActiveValidatorCount++
    k.SetModuleState(ctx, state)

    return nil
}

    ctx sdk.Context,

        return types.ErrInvalidStatus

    validator.ProbationStart = ctx.BlockTime()
    validator.ProbationReason = reason
    k.setAuthorityValidator(ctx, validator)

// RemoveValidator handles MsgRemoveValidator
func (k Keeper) RemoveValidator(
    ctx sdk.Context,
    operatorAddr string,
    reason string,
) error {
    params := k.GetParams(ctx)
    validator, err := k.getAuthorityValidator(ctx, operatorAddr)
    if err != nil {
        return err
    }

    // Enforce composition guarantee: cannot remove if it would drop
    // below minimum for this category
    categoryCount := k.getCategoryCount(ctx, validator.Category)
    minForCategory := k.getMinForCategory(params, validator.Category)
    if categoryCount <= minForCategory {
        return fmt.Errorf(
            "cannot remove: would drop %s below minimum (%d/%d)",
            validator.Category, categoryCount-1, minForCategory,
        )
    }

    // Enforce min_validators
    state := k.GetModuleState(ctx)
    if state.ActiveValidatorCount <= params.MinValidators {
        return types.ErrBelowMinValidators
    }

    validator.Status = "REMOVED"
    k.setAuthorityValidator(ctx, validator)

    state.ActiveValidatorCount--
    k.decrementCategoryCount(ctx, validator.Category)
    k.SetModuleState(ctx, state)

    ctx.EventManager().EmitEvent(
        sdk.NewEvent("validator_removed",
            sdk.NewAttribute("operator", operatorAddr),
            sdk.NewAttribute("reason", reason),
        ),
    )

    return nil
}
```

### Compensation Distribution

```go
// x/authority/keeper/compensation.go

package keeper

import (
    sdk "github.com/cosmos/cosmos-sdk/types"
    "github.com/regen-network/regen-ledger/x/authority/types"
)

// DistributeCompensation splits the validator fund among active validators.
// Called by EndBlocker at each compensation period boundary.
//
// Compensation model: fixed base (equal share) + performance bonus.
// Per Phase 2.6: base = validator_fund / active_count / period,
// bonus pool = performance_bonus_share of total validator fund.
func (k Keeper) DistributeCompensation(ctx sdk.Context) error {
    params := k.GetParams(ctx)

    // Get validator fund balance (from M013 fee routing)
    fundBalance := k.bankKeeper.GetBalance(
        ctx,
        k.accountKeeper.GetModuleAddress(types.ValidatorFundName),
        "uregen",
    )

    if fundBalance.IsZero() {
        return nil // No funds to distribute
    }

    activeValidators := k.getActiveValidators(ctx)
    if len(activeValidators) == 0 {
        return nil
    }

    totalFund := fundBalance.Amount

    // Separate bonus pool
    bonusPool := totalFund.
        Mul(sdk.NewIntFromUint64(uint64(params.PerformanceBonusShareBps))).
        Quo(sdk.NewInt(10000))
    basePool := totalFund.Sub(bonusPool)

    // Base compensation: equal split
    basePerValidator := basePool.Quo(sdk.NewInt(int64(len(activeValidators))))

    // Performance bonus: proportional to composite score
    totalCompositeScore := uint64(0)
    for _, v := range activeValidators {
        if v.Performance != nil {
            totalCompositeScore += uint64(v.Performance.CompositeScore)
        }
    }

    for _, validator := range activeValidators {
        compensation := basePerValidator

        // Add performance bonus (if enabled and scores exist)
        if totalCompositeScore > 0 && bonusPool.IsPositive() && validator.Performance != nil {
            bonus := bonusPool.
                Mul(sdk.NewIntFromUint64(uint64(validator.Performance.CompositeScore))).
                Quo(sdk.NewIntFromUint64(totalCompositeScore))
            compensation = compensation.Add(bonus)
        }

        // Transfer from validator fund to validator operator
        if compensation.IsPositive() {
            valAddr, err := sdk.AccAddressFromBech32(validator.OperatorAddress)
            if err != nil {
                ctx.Logger().Error("invalid validator address", "addr", validator.OperatorAddress)
                continue
            }

            coins := sdk.NewCoins(sdk.NewCoin("uregen", compensation))
            if err := k.bankKeeper.SendCoinsFromModuleToAccount(
                ctx, types.ValidatorFundName, valAddr, coins,
            ); err != nil {
                ctx.Logger().Error("compensation transfer failed",
                    "validator", validator.OperatorAddress, "error", err)
                continue
            }
        }
    }

    ctx.EventManager().EmitEvent(
        sdk.NewEvent("compensation_distributed",
            sdk.NewAttribute("total", totalFund.String()),
            sdk.NewAttribute("validators", fmt.Sprintf("%d", len(activeValidators))),
        ),
    )

    return nil
}
```

```go
    params := k.GetParams(ctx)
    state := k.GetModuleState(ctx)

        )
    }

    // Check composition requirements
    k.checkCompositionRequirements(ctx, params)
}

func (k Keeper) checkCompositionRequirements(ctx sdk.Context, params types.AuthorityParams) {
    state := k.GetModuleState(ctx)

    categories := []struct {
        name  string
        count uint32
        min   uint32
    }{
        {"infrastructure_builder", state.InfrastructureCount, params.MinInfrastructureBuilders},
        {"refi_partner", state.RefiPartnerCount, params.MinRefiPartners},

    for _, cat := range categories {
        if cat.count < cat.min {
            ctx.EventManager().EmitEvent(
                sdk.NewEvent("composition_shortfall",
                    sdk.NewAttribute("category", cat.name),
                    sdk.NewAttribute("current", fmt.Sprintf("%d", cat.count)),
```

### State Transitions

```
States: {CANDIDATE, APPROVED, ACTIVE, PROBATION, REMOVED, TERM_EXPIRED}

CANDIDATE → APPROVED
  trigger: validator_governance.approve(application)
  guard: meets_composition_criteria, slot_available, no_active_conflicts
  action: create AuthorityValidator, schedule activation

APPROVED → ACTIVE
  trigger: node_operational AND signing_key_in_validator_set
  guard: infrastructure_verified
  action: increment active_validator_count, begin block production

ACTIVE → PROBATION
  trigger: performance_below_threshold OR governance_concern_raised
  guard: AGENT-004 performance report OR governance motion
  action: set probation_start, probation_reason; reduce compensation by 50%

PROBATION → ACTIVE
  trigger: performance_restored AND probation_period_elapsed
  guard: AGENT-004 confirms restoration
  action: restore full compensation, clear probation fields

PROBATION → REMOVED
  trigger: probation_period_elapsed AND performance_not_restored
  guard: validator_governance.confirm_removal()
  action: remove from active set, decrement counts, archive record

ACTIVE → TERM_EXPIRED
  trigger: block_time >= term_end
  guard: none
  action: initiate re-application or graceful exit

TERM_EXPIRED → CANDIDATE
  trigger: validator.reapply()
  action: create new application (streamlined for incumbents with good records)

Module Phase Transitions:

  INACTIVE → TRANSITION
    trigger: governance.approve(MsgUpdateAuthorityParams) with initial set
    action: begin parallel operation with PoS

  TRANSITION → ACTIVE
    trigger: active_validator_count >= min_validators AND
             composition requirements met AND
             governance.approve(disable_pos_inflation)
    action: disable PoS staking rewards, full PoA operation

  ACTIVE → EQUILIBRIUM
    trigger: operational for 4+ consecutive quarters without
             composition shortfalls or emergency events
    action: reduce monitoring frequency, enable streamlined re-approval
```

### Security Invariants

1. **Composition Guarantee**: Active set must maintain `min_infrastructure_builders`, `min_refi_partners`, and `min_data_stewards` at all times; removal blocked if it would violate
2. **Byzantine Tolerance**: Active set > 3f + 1 where f = maximum tolerated Byzantine validators (Tendermint standard)
3. **No Self-Approval**: Existing active validators cannot unilaterally approve new validators; requires governance proposal process
4. **Term Accountability**: No validator may serve beyond `term_end` without re-approval
5. **Compensation Cap**: Total validator compensation ≤ validator fund balance; no inflationary fallback
6. **Graceful Degradation**: If active set < `min_validators`, P0 emergency event emitted; AGENT-004 and core team notified

## Native Module: x/rewards (M015)

### Overview

The `x/rewards` module replaces passive staking rewards with activity-based distribution from the Community Pool. Participants earn proportional to their ecological and governance contributions, with an optional stability tier for long-term holders seeking predictable returns. The module tracks on-chain activity, computes per-address scores, and distributes rewards at each epoch boundary.

```go
// x/rewards/keeper/distribution.go

package keeper

import (
    sdk "github.com/cosmos/cosmos-sdk/types"
    "github.com/regen-network/regen-ledger/x/rewards/types"
)

// DistributeRewards performs per-period reward distribution.
// Called from EndBlocker when a period boundary is crossed.
func (k Keeper) DistributeRewards(ctx sdk.Context) error {
    params := k.GetRewardParams(ctx)
    state := k.GetModuleState(ctx)

    if state.Phase != "DISTRIBUTING" {
        return nil
    }

    // 1. Determine Community Pool inflow for this period
    communityPoolInflow := k.getCommunityPoolInflowForPeriod(ctx)
    if communityPoolInflow.IsZero() {
        return nil // No revenue to distribute
    }

    // 2. Stability Tier Allocation (first priority, capped at max_stability_share)
    totalStabilityCommitted, _ := sdk.NewIntFromString(state.TotalStabilityCommitted)
    stabilityReturn := totalStabilityCommitted.
        Mul(sdk.NewIntFromUint64(uint64(params.StabilityTierAnnualReturnBps))).
        Quo(sdk.NewInt(10000)).
        Quo(sdk.NewInt(int64(365 * 86400 / params.DistributionPeriodSeconds)))

    maxStability := communityPoolInflow.
        Mul(sdk.NewIntFromUint64(uint64(params.MaxStabilityShareBps))).
        Quo(sdk.NewInt(10000))

    stabilityAllocation := sdk.MinInt(stabilityReturn, maxStability)

    // 3. Remaining pool for activity-based distribution
    activityPool := communityPoolInflow.Sub(stabilityAllocation)

    // 4. Distribute stability tier rewards
    if stabilityAllocation.IsPositive() {
        k.distributeStabilityRewards(ctx, stabilityAllocation)
    }

    // 5. Compute activity scores and distribute
    if activityPool.IsPositive() {
        k.distributeActivityRewards(ctx, params, activityPool)
    }

    // 6. Record distribution
    state.CurrentPeriod++
    state.TotalRewardsDistributed = sdk.NewIntFromString(state.TotalRewardsDistributed).
        Add(stabilityAllocation).Add(activityPool).String()
    k.SetModuleState(ctx, state)

    k.recordDistribution(ctx, state.CurrentPeriod, communityPoolInflow,
        stabilityAllocation, activityPool)

    ctx.EventManager().EmitEvent(
        sdk.NewEvent("rewards_distributed",
            sdk.NewAttribute("period", fmt.Sprintf("%d", state.CurrentPeriod)),
            sdk.NewAttribute("stability", stabilityAllocation.String()),
            sdk.NewAttribute("activity", activityPool.String()),
        ),
    )

    return nil
}

// distributeActivityRewards computes per-address scores and distributes
// proportionally from the activity pool.
func (k Keeper) distributeActivityRewards(
    ctx sdk.Context,
    params types.RewardParams,
    activityPool sdk.Int,
) {
    scores := k.getAllActivityScoresForPeriod(ctx)

    // Compute composite scores
    totalScore := sdk.ZeroDec()
    compositeScores := make(map[string]sdk.Dec)

    for _, score := range scores {
        composite := k.computeCompositeScore(ctx, score, params)
        compositeScores[score.Participant] = composite
        totalScore = totalScore.Add(composite)
    }

    if totalScore.IsZero() {
        return
    }

    // Distribute proportionally
    for participant, composite := range compositeScores {
        share := composite.Quo(totalScore)
        reward := share.MulInt(activityPool).TruncateInt()

        if reward.IsPositive() {
            addr, err := sdk.AccAddressFromBech32(participant)
            if err != nil {
                continue
            }

            coins := sdk.NewCoins(sdk.NewCoin("uregen", reward))
            if err := k.bankKeeper.SendCoinsFromModuleToAccount(
                ctx, types.ModuleName, addr, coins,
            ); err != nil {
                ctx.Logger().Error("reward distribution failed",
                    "participant", participant, "error", err)
                continue
            }
        }
    }
}

// computeCompositeScore calculates a participant's weighted activity score
// per the Phase 2.6 formula.
//
// Note: This runs inside EndBlocker via DistributeRewards, so it must never
// panic. Malformed score values are treated as zero with a log warning.
func (k Keeper) computeCompositeScore(
    ctx sdk.Context,
    score types.ActivityScore,
    params types.RewardParams,
) sdk.Dec {
    purchaseVal, err := sdk.NewDecFromStr(score.PurchaseValue)
    if err != nil {
        ctx.Logger().Error("invalid purchase value in activity score",
            "participant", score.Participant, "value", score.PurchaseValue, "error", err)
        purchaseVal = sdk.ZeroDec()
    }
    retireVal, err := sdk.NewDecFromStr(score.RetirementValue)
    if err != nil {
        ctx.Logger().Error("invalid retirement value in activity score",
            "participant", score.Participant, "value", score.RetirementValue, "error", err)
        retireVal = sdk.ZeroDec()
    }
    facilitateVal, err := sdk.NewDecFromStr(score.FacilitationValue)
    if err != nil {
        ctx.Logger().Error("invalid facilitation value in activity score",
            "participant", score.Participant, "value", score.FacilitationValue, "error", err)
        facilitateVal = sdk.ZeroDec()
    }
    govVotes := sdk.NewDec(int64(score.GovernanceVotes))

    // Proposal score with anti-gaming: full weight for passed, half for
    // quorum-but-failed, zero for no quorum (per Phase 2.6 spec)
    proposalScore := sdk.NewDec(int64(score.ProposalsPassed)).
        Add(sdk.NewDec(int64(score.ProposalsQuorumReached - score.ProposalsPassed)).
            Mul(sdk.NewDecWithPrec(5, 1))) // 0.5 weight

    composite := purchaseVal.
        Mul(sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.WeightPurchase)))).
        Add(retireVal.
            Mul(sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.WeightRetirement))))).
        Add(facilitateVal.
            Mul(sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.WeightFacilitation))))).
        Add(govVotes.
            Mul(sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.WeightGovernance))))).
        Add(proposalScore.
            Mul(sdk.NewDecFromInt(sdk.NewIntFromUint64(uint64(params.WeightProposals)))))

    return composite.Quo(sdk.NewDec(10000)) // Normalize from basis points
}
```

```go
    "time"

    sdk "github.com/cosmos/cosmos-sdk/types"
    "github.com/regen-network/regen-ledger/x/rewards/types"
)

// CommitStability handles MsgCommitStability
func (k Keeper) CommitStability(
    ctx sdk.Context,
    msg *types.MsgCommitStability,
) (*types.MsgCommitStabilityResponse, error) {
    params := k.GetRewardParams(ctx)

    // Validate lock period
    if msg.LockPeriodDays < params.MinLockPeriodDays ||
        msg.LockPeriodDays > params.MaxLockPeriodDays {
        return nil, types.ErrInvalidLockPeriod
    }

    // Validate minimum amount
    minAmount, _ := sdk.NewIntFromString(params.MinStabilityAmount)
    if msg.Amount.Amount.LT(minAmount) {
        return nil, types.ErrInsufficientAmount
    }

    holder, err := sdk.AccAddressFromBech32(msg.Holder)
    if err != nil {
        return nil, err
    }

    // Lock tokens in module account
    if err := k.bankKeeper.SendCoinsFromAccountToModule(
        ctx, holder, types.StabilityPoolName, sdk.NewCoins(msg.Amount),
    ); err != nil {
        return nil, err
    }

    // Create commitment
    maturity := ctx.BlockTime().Add(
        time.Duration(msg.LockPeriodDays) * 24 * time.Hour,
    )

    commitID := k.getNextCommitmentID(ctx)
    commitment := types.StabilityCommitment{
        Id:           commitID,
        Holder:       msg.Holder,
        Amount:       msg.Amount,
        LockPeriodDays: msg.LockPeriodDays,
        CommittedAt:  ctx.BlockTime(),
        MaturityDate: maturity,
        AccruedRewards: "0",
        Status:       "ACTIVE",
    }

    k.setStabilityCommitment(ctx, &commitment)

    // Update total committed
    state := k.GetModuleState(ctx)
    totalCommitted, _ := sdk.NewIntFromString(state.TotalStabilityCommitted)
    state.TotalStabilityCommitted = totalCommitted.Add(msg.Amount.Amount).String()
    k.SetModuleState(ctx, state)

    // Estimate annual reward
    annualReward := msg.Amount.Amount.
        Mul(sdk.NewIntFromUint64(uint64(params.StabilityTierAnnualReturnBps))).
        Quo(sdk.NewInt(10000))

    ctx.EventManager().EmitEvent(
        sdk.NewEvent("stability_committed",
            sdk.NewAttribute("commitment_id", fmt.Sprintf("%d", commitID)),
            sdk.NewAttribute("holder", msg.Holder),
            sdk.NewAttribute("amount", msg.Amount.String()),
            sdk.NewAttribute("maturity", maturity.String()),
        ),
    )

    return &types.MsgCommitStabilityResponse{
        CommitmentId:          commitID,
        MaturityDate:          maturity,
        EstimatedAnnualReward: annualReward.String(),
    }, nil
}

// ExitStabilityEarly handles MsgExitStabilityEarly
func (k Keeper) ExitStabilityEarly(
    ctx sdk.Context,
    msg *types.MsgExitStabilityEarly,
) (*types.MsgExitStabilityEarlyResponse, error) {
    params := k.GetRewardParams(ctx)

    commitment, err := k.getStabilityCommitment(ctx, msg.CommitmentId)
    if err != nil {
        return nil, err
    }

    if commitment.Holder != msg.Holder {
        return nil, types.ErrUnauthorized
    }

    if commitment.Status != "ACTIVE" {
        return nil, types.ErrInvalidCommitmentStatus
    }

    holder, err := sdk.AccAddressFromBech32(msg.Holder)
    if err != nil {
        return nil, err
    }

    // Return locked tokens
    if err := k.bankKeeper.SendCoinsFromModuleToAccount(
        ctx, types.StabilityPoolName, holder, sdk.NewCoins(commitment.Amount),
    ); err != nil {
        return nil, err
    }

    // Calculate forfeited rewards (50% of accrued)
    accrued, _ := sdk.NewIntFromString(commitment.AccruedRewards)
    penalty := accrued.
        Mul(sdk.NewIntFromUint64(uint64(params.EarlyExitPenaltyBps))).
        Quo(sdk.NewInt(10000))
    returned := accrued.Sub(penalty)

    // Pay non-forfeited rewards
    if returned.IsPositive() {
        if err := k.bankKeeper.SendCoinsFromModuleToAccount(
            ctx, types.ModuleName, holder,
            sdk.NewCoins(sdk.NewCoin("uregen", returned)),
        ); err != nil {
            return nil, err
        }
    }

    // Update commitment
    commitment.Status = "EARLY_EXIT"
    k.setStabilityCommitment(ctx, commitment)

    // Update total committed
    state := k.GetModuleState(ctx)
    totalCommitted, _ := sdk.NewIntFromString(state.TotalStabilityCommitted)
    state.TotalStabilityCommitted = totalCommitted.Sub(commitment.Amount.Amount).String()
    k.SetModuleState(ctx, state)

    return &types.MsgExitStabilityEarlyResponse{
        Returned:         sdk.NewCoin("uregen", returned),
        ForfeitedRewards: sdk.NewCoin("uregen", penalty),
    }, nil
}
```

### State Transitions

```
Module Phase Transitions:

  INACTIVE → TRACKING
    trigger: governance.approve(MsgActivateRewards)
    guard: M013 active (Community Pool receiving fee revenue)
    action: begin recording activity scores per address per period;
            set tracking_ends_at = now + tracking_period_seconds

  TRACKING → DISTRIBUTING
    trigger: block_time >= tracking_ends_at AND sufficient_data
    guard: activity_score data validated, no anomalies flagged
    action: begin automatic distribution at end of each period

Stability Tier Lifecycle:

  UNCOMMITTED → ACTIVE
    trigger: holder.commit_stability(amount, lock_period)
    guard: amount >= min_stability_amount, lock_period in [min, max]
    action: lock tokens in stability pool, begin accruing returns

  ACTIVE → MATURED
    trigger: block_time >= maturity_date
    action: unlock tokens + accrued rewards; holder may re-commit

  ACTIVE → EARLY_EXIT
    trigger: holder.exit_early()
    action: unlock tokens, forfeit early_exit_penalty_bps of accrued rewards
```

### Security Invariants

1. **Revenue Constraint**: Total distributions per period ≤ Community Pool inflow for that period (no deficit spending)
2. **Stability Cap**: Stability tier allocation ≤ `max_stability_share_bps` of Community Pool inflow
3. **Activity Integrity**: Activity scores derived exclusively from on-chain transactions; no self-reported activity
4. **Sybil Resistance**: Activity weights ensure wash trading is unprofitable (M013 fees paid > rewards earned for circular transactions)
5. **No Double-Counting**: Each transaction counted once. Marketplace trades: credit assigned to buyer. Retirements: credit assigned to retiree.
6. **Governance Override**: All distribution parameters changeable via Layer 3 governance at any time
7. **Weight Sum Unity**: `weight_purchase + weight_retirement + weight_facilitation + weight_governance + weight_proposals = 10000` (enforced on write)
