## CosmWasm Contract Deployment

### Deployment Script

```bash
#!/bin/bash
# scripts/deploy-contracts-testnet.sh

set -euo pipefail

CHAIN_ID="regen-testnet-1"
NODE="http://localhost:26657"
DEPLOYER="regen1deployer..."
GAS_PRICES="0.025uregen"

echo "=== CosmWasm Contract Deployment to Testnet ==="

# 1. Store contract bytecodes
echo "Storing attestation-bond contract..."
ATTEST_CODE_ID=$(regen tx wasm store \
  artifacts/attestation_bond.wasm \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="store_code") | .attributes[] | select(.key=="code_id") | .value')
echo "attestation-bond code ID: $ATTEST_CODE_ID"

echo "Storing service-escrow contract..."
ESCROW_CODE_ID=$(regen tx wasm store \
  artifacts/service_escrow.wasm \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="store_code") | .attributes[] | select(.key=="code_id") | .value')
echo "service-escrow code ID: $ESCROW_CODE_ID"

echo "Storing reputation-registry contract..."
REPUTATION_CODE_ID=$(regen tx wasm store \
  artifacts/reputation_registry.wasm \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="store_code") | .attributes[] | select(.key=="code_id") | .value')
echo "reputation-registry code ID: $REPUTATION_CODE_ID"

echo "Storing marketplace-curation contract..."
CURATION_CODE_ID=$(regen tx wasm store \
  artifacts/marketplace_curation.wasm \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="store_code") | .attributes[] | select(.key=="code_id") | .value')
echo "marketplace-curation code ID: $CURATION_CODE_ID"

# 2. Deploy DAO DAO Arbiter DAO (prerequisite for M008 and M009)
echo "Deploying Arbiter DAO..."
ARBITER_DAO_ADDR=$(regen tx wasm instantiate $DAO_DAO_CODE_ID \
  "$(cat config/arbiter-dao-testnet.json)" \
  --label "arbiter-dao-testnet" --admin $DEPLOYER \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="instantiate") | .attributes[] | select(.key=="_contract_address") | .value')
echo "Arbiter DAO address: $ARBITER_DAO_ADDR"

# 3. Instantiate contracts
echo "Instantiating attestation-bond..."
ATTEST_ADDR=$(regen tx wasm instantiate $ATTEST_CODE_ID \
  "{\"arbiter_dao\":\"$ARBITER_DAO_ADDR\",\"min_challenge_deposit_ratio\":\"1000\",\"challenge_period_days\":7,\"arbiter_fee_ratio\":\"500\",\"attestation_types\":[{\"name\":\"project_boundary\",\"min_bond\":\"500000000\",\"lock_period_days\":90,\"challenge_window_days\":60},{\"name\":\"baseline_measurement\",\"min_bond\":\"1000000000\",\"lock_period_days\":180,\"challenge_window_days\":120},{\"name\":\"credit_issuance_claim\",\"min_bond\":\"2000000000\",\"lock_period_days\":365,\"challenge_window_days\":300}]}" \
  --label "attestation-bond-testnet" --admin $DEPLOYER \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="instantiate") | .attributes[] | select(.key=="_contract_address") | .value')
echo "attestation-bond address: $ATTEST_ADDR"

echo "Instantiating reputation-registry..."
REPUTATION_ADDR=$(regen tx wasm instantiate $REPUTATION_CODE_ID \
  "{\"staking_contract\":\"$STAKING_CONTRACT\",\"decay_half_life_days\":365,\"min_stake_by_category\":[[\"credit_class_quality\",\"1000000000\"],[\"project_legitimacy\",\"500000000\"],[\"verifier_competence\",\"2000000000\"],[\"methodology_rigor\",\"5000000000\"]]}" \
  --label "reputation-registry-testnet" --admin $DEPLOYER \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="instantiate") | .attributes[] | select(.key=="_contract_address") | .value')
echo "reputation-registry address: $REPUTATION_ADDR"

echo "Instantiating service-escrow..."
ESCROW_ADDR=$(regen tx wasm instantiate $ESCROW_CODE_ID \
  "{\"arbiter_dao\":\"$ARBITER_DAO_ADDR\",\"provider_bond_ratio\":\"1000\",\"default_review_period_days\":14}" \
  --label "service-escrow-testnet" --admin $DEPLOYER \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="instantiate") | .attributes[] | select(.key=="_contract_address") | .value')

echo "Instantiating marketplace-curation..."

  "{\"reputation_registry\":\"$REPUTATION_ADDR\",\"min_curation_bond\":\"1000000000\",\"listing_fee\":\"10000000\",\"curation_fee_rate\":\"50\",\"min_quality_score\":300}" \
  --label "marketplace-curation-testnet" --admin $DEPLOYER \
  --from $DEPLOYER --gas auto --gas-adjustment 1.3 \
  --gas-prices $GAS_PRICES --chain-id $CHAIN_ID \
  --node $NODE -y --output json | jq -r '.logs[0].events[] | select(.type=="instantiate") | .attributes[] | select(.key=="_contract_address") | .value')
echo "marketplace-curation address: $CURATION_ADDR"

# 4. Save deployment addresses
cat > deployment-testnet.json << EOF
{
  "chain_id": "$CHAIN_ID",
  "deployed_at": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "contracts": {
    "arbiter_dao": "$ARBITER_DAO_ADDR",
    "attestation_bond": "$ATTEST_ADDR",
    "reputation_registry": "$REPUTATION_ADDR",
    "service_escrow": "$ESCROW_ADDR",
    "marketplace_curation": "$CURATION_ADDR"
  },
  "code_ids": {
    "attestation_bond": $ATTEST_CODE_ID,
    "service_escrow": $ESCROW_CODE_ID,
    "reputation_registry": $REPUTATION_CODE_ID,
    "marketplace_curation": $CURATION_CODE_ID
  }
}
EOF

echo "=== Deployment Complete ==="
echo "Addresses saved to deployment-testnet.json"
```

#!/bin/bash
# scripts/deploy-agents-staging.sh

set -euo pipefail

echo "=== Agent Staging Deployment ==="

# 1. Apply database migrations
echo "Running database migrations..."
psql $DATABASE_URL -f sql/init.sql
psql $DATABASE_URL -f sql/migrations/001_agent_tables.sql
psql $DATABASE_URL -f sql/migrations/002_workflow_tables.sql
psql $DATABASE_URL -f sql/migrations/003_economic_reboot_tables.sql

# 2. Generate agent keypairs (Ed25519)
echo "Generating agent keypairs..."
for AGENT in agent-001 agent-002 agent-003 agent-004; do
    if [ ! -f "secrets/${AGENT}.key" ]; then
        regen keys add $AGENT --keyring-backend test --output json > "secrets/${AGENT}.key"
        echo "Generated key for $AGENT"
    else
        echo "Key exists for $AGENT, skipping"
    fi
done

# 3. Fund agent accounts from testnet faucet
echo "Funding agent accounts..."
for AGENT in agent-001 agent-002 agent-003 agent-004; do
    ADDR=$(cat "secrets/${AGENT}.key" | jq -r '.address')
    curl -s -X POST "http://faucet.testnet.regen.network/faucet" \
      -H "Content-Type: application/json" \
      -d "{\"address\": \"$ADDR\", \"coins\": [\"100000000000uregen\"]}"

done

# 4. Register agents with the system
echo "Registering agents..."

  -H "Content-Type: application/json" \

    "agents": [

## Smoke Test Procedures

### Automated Smoke Test Suite

```bash
#!/bin/bash
# scripts/smoke-test-testnet.sh

set -euo pipefail

PASSED=0
FAILED=0

check() {
    local name="$1"
    local result="$2"
    if [ "$result" = "true" ]; then
        echo "✓ $name"
        PASSED=$((PASSED + 1))
    else
        echo "✗ $name"
        FAILED=$((FAILED + 1))
    fi
}

echo "=== Testnet Smoke Tests ==="
echo ""

# --- Chain Health ---
echo "## Chain Health"
LATEST_HEIGHT=$(regen query block latest --node $NODE 2>/dev/null | jq -r '.block.header.height')
check "Chain producing blocks" "$([ "$LATEST_HEIGHT" -gt 0 ] && echo true || echo false)"

VALIDATORS=$(regen query staking validators --node $NODE --output json | jq '.validators | length')
check "Validators active" "$([ "$VALIDATORS" -gt 0 ] && echo true || echo false)"

# --- MCP Services ---
echo ""
echo "## MCP Services"
LEDGER_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:3001/health)
check "Ledger MCP healthy" "$([ "$LEDGER_STATUS" = "200" ] && echo true || echo false)"

KOI_STATUS=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:3002/health)
check "KOI MCP healthy" "$([ "$KOI_STATUS" = "200" ] && echo true || echo false)"

# --- Contracts ---
echo ""
echo "## Smart Contracts"
DEPLOYMENT=$(cat deployment-testnet.json)

ATTEST_INFO=$(regen query wasm contract $(echo $DEPLOYMENT | jq -r '.contracts.attestation_bond') --node $NODE 2>/dev/null)
check "Attestation bond deployed" "$([ -n "$ATTEST_INFO" ] && echo true || echo false)"

REPUTATION_INFO=$(regen query wasm contract $(echo $DEPLOYMENT | jq -r '.contracts.reputation_registry') --node $NODE 2>/dev/null)
check "Reputation registry deployed" "$([ -n "$REPUTATION_INFO" ] && echo true || echo false)"

ESCROW_INFO=$(regen query wasm contract $(echo $DEPLOYMENT | jq -r '.contracts.service_escrow') --node $NODE 2>/dev/null)
check "Service escrow deployed" "$([ -n "$ESCROW_INFO" ] && echo true || echo false)"

CURATION_INFO=$(regen query wasm contract $(echo $DEPLOYMENT | jq -r '.contracts.marketplace_curation') --node $NODE 2>/dev/null)
check "Marketplace curation deployed" "$([ -n "$CURATION_INFO" ] && echo true || echo false)"

# --- Agents ---
echo ""
echo "## Agent Health"
AGENT_STATUS=$(curl -s http://localhost:3000/api/agents)
for AGENT_NAME in RegenRegistryReviewer RegenGovernanceAnalyst RegenMarketMonitor RegenValidatorMonitor; do
    STATUS=$(echo $AGENT_STATUS | jq -r ".agents[] | select(.name==\"$AGENT_NAME\") | .status")
    check "$AGENT_NAME active" "$([ "$STATUS" = "active" ] && echo true || echo false)"
done

# --- Native Modules (if deployed) ---
echo ""
echo "## Native Modules"
FEE_STATE=$(regen query feerouter module-state --node $NODE --output json 2>/dev/null || echo '{"phase":"not_deployed"}')
FEE_PHASE=$(echo $FEE_STATE | jq -r '.phase')
check "Fee router module" "$([ "$FEE_PHASE" != "not_deployed" ] && echo true || echo false)"

AUTHORITY_STATE=$(regen query authority module-state --node $NODE --output json 2>/dev/null || echo '{"phase":"not_deployed"}')
AUTH_PHASE=$(echo $AUTHORITY_STATE | jq -r '.phase')
check "Authority module" "$([ "$AUTH_PHASE" != "not_deployed" ] && echo true || echo false)"

# --- Functional Tests ---
echo ""
echo "## Functional Tests"

# Submit a test reputation signal
SIGNAL_RESULT=$(regen tx wasm execute $(echo $DEPLOYMENT | jq -r '.contracts.reputation_registry') \
  '{"submit_signal":{"subject_type":"credit_class","subject_id":"C01","category":"credit_class_quality","endorsement_level":4}}' \
  --from agent-001 --gas auto --gas-adjustment 1.3 \
  --gas-prices 0.025uregen --chain-id $CHAIN_ID \
  --node $NODE -y --output json 2>/dev/null | jq -r '.code // "1"')
check "Reputation signal submission" "$([ "$SIGNAL_RESULT" = "0" ] && echo true || echo false)"

# Query reputation score
SCORE_RESULT=$(regen query wasm contract-state smart $(echo $DEPLOYMENT | jq -r '.contracts.reputation_registry') \
  '{"aggregated_reputation":{"subject_type":"CreditClass","subject_id":"C01","category":"credit_class_quality"}}' \
  --node $NODE --output json 2>/dev/null | jq -r '.data.score // "0"')
check "Reputation score computed" "$([ "$SCORE_RESULT" != "0" ] && echo true || echo false)"

# --- Summary ---
echo ""
echo "=== Results ==="
echo "Passed: $PASSED"
echo "Failed: $FAILED"
echo "Total:  $((PASSED + FAILED))"

if [ $FAILED -gt 0 ]; then
    echo ""
    echo "SMOKE TESTS FAILED — review failures before proceeding"
    exit 1
fi

echo ""
echo "ALL SMOKE TESTS PASSED"
```
